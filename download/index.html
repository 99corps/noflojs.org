To use NoFlo in your web app, simply include it to your pages: 

<script src="/path/to/noflo.js"></script>
<script>
var noflo = require('/noflo');
noflo.graph.loadJSON(someJSON, function (graph) {
  graph.baseDir = '/noflo-custom';
  noflo.createNetwork(graph, function (network) {
    console.log("Network created");
  });
});
</script>
